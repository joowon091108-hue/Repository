function rollAll() { const diceElements = document.querySelectorAll('.dice'); rollButton.disabled = true; let total = 0; // 합계 저장 변수 diceElements.forEach((dice, index) => { dice.classList.remove('rolling'); void dice.offsetWidth; dice.classList.add('rolling'); setTimeout(() => { const result = Math.floor(Math.random() * 6) + 1; dice.innerText = result; total += result; // 나온 숫자를 더함 if (index === diceElements.length - 1) { rollButton.disabled = false; // 제목에 합계를 표시하거나 alert로 띄워줄 수 있어! document.querySelector('h2').innerText = "합계: " + total; } }, 400); }); }
